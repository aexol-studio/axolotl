# Resolvers

Create a `resolvers.ts` file. Resolver arguments are typeâ€‘safe; you can also
use the TypeScript `satisfies` operator with types from `models.ts` to validate
responses.

```ts filename="resolvers.ts"
import { BeerOrm } from '@/src/ormBeersFile.js';
import { createResolvers } from '@/src/axolotl.js'

const Beer = BeerOrm();

export default createResolvers({
  Query: {
    beers: () => Beer.list(),
  },
  Mutation: {
    // the type of input and args is inferred ğŸ¥³ from createResolvers function
    addBeer: (input, args) => {
      return Beer.create(args.beer);
    },
    deleteBeer: (input, args) => {
      return Beer.remove(args);
    },
    updateBeer: (input, args) => {
      return Beer.update(args._id, args.beer);
    },
  },
});
```
